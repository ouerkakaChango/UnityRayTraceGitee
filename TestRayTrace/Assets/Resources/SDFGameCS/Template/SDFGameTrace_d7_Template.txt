#define SSAA_num 1
[numthreads(8, 8, 1)]
void Render(uint3 id : SV_DispatchThreadID)
{

	int idx = id.x;
	int idy = id.y;

	int SSAA_edge = (int)sqrt(SSAA_num);
	float3 pixLeftDownPos = screenLeftDownPix + screenU * pixW * (idx - 0.5) + screenV * pixH * (idy - 0.5);
	float SSAA_unitDis = 1.0f / SSAA_edge;
	float3 SSAA_colors[SSAA_num];
	float3 SSAA_indirColors[SSAA_num];
	float SSAA_eyeDepth[SSAA_num];

	float3 camFoward = normalize(cross(screenU,screenV));

	seed = float3(id.xy,frameID);
	suv = float2(id.x/(float)w,id.y/(float)h);

	[unroll(SSAA_edge)]
	for (int j = 1; j <= SSAA_edge; j++)
	{
		[unroll(SSAA_edge)]
		for (int i = 1; i <= SSAA_edge; i++)
		{
			Ray ray;
			ray.pos = pixLeftDownPos + screenU * pixW * i * SSAA_unitDis + screenV * pixH * j * SSAA_unitDis;
			if(camType == 1)
			{
				ray.dir = camFoward;
			}
			else
			{
				ray.dir = normalize(ray.pos - eyePos);
			}
			HitInfo minHit;
			Material_PBR mat;
			SSAA_colors[i - 1 + (j - 1)* SSAA_edge] = GetSDFGameColor(ray, minHit, mat);
			if(useIndirectRT)
			{
				SSAA_indirColors[i - 1 + (j - 1)* SSAA_edge] = GetSDFGameIndirColor(ray,seed,minHit,mat);
			}
			if(needEyeDepth)
			{
				float eyeDepth = -1;
				if(minHit.bHit)
				{
					eyeDepth = length(minHit.P - ray.pos);
				}
				SSAA_eyeDepth[i - 1 + (j - 1)* SSAA_edge] = eyeDepth;
			}
		}
	}

	float3 re = 0;
	for (int i = 0; i < SSAA_num; i++)
	{
		re += SSAA_colors[i];
	}
	re /= (float)SSAA_num;
	

	float eyeDepth = 0;
	int cc = 0;
	if(needEyeDepth)
	{
		for (int i = 0; i < SSAA_num; i++)
		{
		if(SSAA_eyeDepth[i]>=0)
		{
			eyeDepth += SSAA_eyeDepth[i];
			cc+=1;
			}
		}
		eyeDepth /= (float)SSAA_num;
		if(cc==0){eyeDepth = MAXFLOAT;}
		rt_EyeDepth[id.xy] = eyeDepth;
	}

	if(useIndirectRT)
	{
		re = 0;
		for (int i = 0; i < SSAA_num; i++)
		{
			re += SSAA_indirColors[i];
		}
		re /= (float)SSAA_num;
		IndirectResult[id.xy] = float4(re, 1);
	}

	if(useQuadTrace)
	{
	float4 SSAA_quadColor[SSAA_num];
		[unroll(SSAA_edge)]
	for (int j = 1; j <= SSAA_edge; j++)
	{
		[unroll(SSAA_edge)]
		for (int i = 1; i <= SSAA_edge; i++)
		{
			Ray ray;
			ray.pos = pixLeftDownPos + screenU * pixW * i * SSAA_unitDis + screenV * pixH * j * SSAA_unitDis;
			if(camType == 1)
			{
				ray.dir = camFoward;
			}
			else
			{
				ray.dir = normalize(ray.pos - eyePos);
			}
			HitInfo minHit;
			Material_PBR mat;
			SSAA_quadColor[i - 1 + (j - 1)* SSAA_edge] = GetQuadTraceColor(ray, minHit, mat);
			{
				float eyeDepth = -1;
				if(minHit.bHit)
				{
					eyeDepth = length(minHit.P - ray.pos);
				}
				SSAA_eyeDepth[i - 1 + (j - 1)* SSAA_edge] = eyeDepth;
			}
		}
	}

	float4 quadColor = 0;
	for (int i = 0; i < SSAA_num; i++)
	{
		quadColor += SSAA_quadColor[i];
	}
	quadColor /= (float)SSAA_num;

		float quadEyeDepth=0;
		cc = 0;
		for (int i = 0; i < SSAA_num; i++)
		{
			if(SSAA_eyeDepth[i]>=0)
			{
			quadEyeDepth += SSAA_eyeDepth[i];
			cc+=1;
			}
		}
		quadEyeDepth /= (float)SSAA_num;
		if(cc == 0 )
		{
			quadEyeDepth = -1;
		}
		if(quadEyeDepth>=0 && quadEyeDepth < eyeDepth)
		{
			//???
			re = quadColor.rgb;
		}
	}

	SmoothWithDither(re, suv);
	Result[id.xy] = float4(re, 1);
}
